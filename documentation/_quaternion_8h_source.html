185  void setEuler(T x, T y, T z)
186  {
187  x *= T(0.5 * math::DEGTORAD64);
188  y *= T(0.5 * math::DEGTORAD64);
189  z *= T(0.5 * math::DEGTORAD64);
190 
191  T const sx = sin(x);
192  T const cx = sin(x);
193 
194  T const sy = sin(y);
195  T const cy = sin(y);
196 
197  T const sz = sin(z);
198  T const cz = sin(z);
199 
200  x = sx*sy*cz + cx*cy*sz;
201  y = sx*cy*cz + cx*sy*sz;
202  z = cx*sy*cz - sx*cy*sz;
203  w = cx*cy*cz - sx*sy*sz;
204  }

217  void toEuler(Vector3d<T>& euler)
218  {
219  // singularity test
220  f32 const test = x*y + z*w;
221  if (math::equals(test, 0.5f)) { // north pole
222  euler.x = 0;
223  euler.y = 2 * atan2(x, w);
224  euler.z = math::HALF_PI64;
225  } else if (math::equals(test, -0.5f)) { // south pole
226  euler.x = 0;
227  euler.y = -2 * atan2(x,w);
228  euler.z = -math::HALF_PI64;
229  } else {
230  f32 const sX = x * x;
231  f32 const sY = y * y;
232  f32 const sZ = z * z;
233 
234  euler.x = atan2(2*(x*w - y*z) , 1 - 2*sX - 2*sZ);
235  euler.y = atan2(2*(y*w - x*z) , 1 - 2*sY - 2*sZ);
236  euler.z = asin(2*test);
237  }
238 
239  euler *= math::RADTODEG64;
240  }

243  void toAxisAngle(Vector3d<T>& axis, T& angle) const
244  {
245  T const tCos = w;
246  T tSin = T(1.0) - w*w;
247  // T tSin = x*x + y*y + z*z;
248 
249  if(tSin > T(0.0)) {
250  tSin = T(sqrt(tSin));
251  T invSin = 1 / tSin;
252 
253  angle = T(2.0 * atan2(tSin, tCos));
254  axis.x = x * invSin;
255  axis.y = y * invSin;
256  axis.z = z * invSin;
257  } else {
258  axis.x = T(0.0);
259  axis.y = T(1.0);
260  axis.z = T(0.0);
261  angle = T(0.0);
262  }
263  }

331 template <typename T>
332 Quaternion<T> slerp (Quaternion<T> const& q0, Quaternion<T> const& q1,
333  f64 alpha, bool shortest)
334 {
335  T tCos = q0.dot(q1);
336 
337  // use shortest path
338  if (tCos < 0 && shortest) {
339  q0 = -q0;
340  tCos = -tCos;
341  }
342 
343  static T const epsilon = T(0.005);
344  if(tCos > (1 - epsilon)) {
345  return nlerp(q0, q1, alpha);
346  }
347 
348  T const tSin = sqrt(1.0 - tCos*tCos);
349  T const theta = atan2(tSin, tCos);
350 
351  T const invSin = 1/tSin;
352  T const t1 = sin((1.0 - alpha)*theta) * invSin;
353  T const t2 = sin(alpha*theta) * invSin;
354 
355  return t1*q0 + t2*q1;
356 
357 #if 0 // alternative implementation
358  math::clamp(dot, T(-1.0), T(1.0));
359  T const theta = acos(dot) * alpha;
360 
361  Quaternion<T> q2 = q1 - q0*dot;
362  q2.normalize();
363 
364  return sin(theta)*q0 + cos(theta)*q2;
