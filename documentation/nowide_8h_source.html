29 template<typename CharOut,typename CharIn>
30 std::basic_string<CharOut>
31 utf_to_utf(CharIn const *begin,CharIn const *end, method_type how = default_method)
32 {
33  std::basic_string<CharOut> result;
34  result.reserve(end-begin);
35  typedef std::back_insert_iterator<std::basic_string<CharOut> > inserter_type;
36  inserter_type inserter(result);
37  code_point c;
38  while(begin!=end) {
39  c=utf_traits<CharIn>::template decode<CharIn const *>(begin,end);
40  if(c==illegal || c==incomplete) {
41  if(how==stop) {
42  break;
43  //throw conversion_error();
44  }
45  } else {
46  utf_traits<CharOut>::template encode<inserter_type>(c,inserter);
47  }
48  }
49  return result;
50 }

72 template<typename CharOut,typename CharIn>
73 CharOut *basic_convert(CharOut *buffer,size_t buffer_size,CharIn const *source_begin,CharIn const *source_end)
74 {
75  CharOut *rv = buffer;
76  if(buffer_size == 0) {
77  return 0;
78  }
79  buffer_size --;
80  while(source_begin!=source_end) {
81  code_point c = utf_traits<CharIn>::template decode<CharIn const *>(source_begin,source_end);
82  if(c==illegal || c==incomplete) {
83  rv = 0;
84  break;
85  }
86  size_t width = utf_traits<CharOut>::width(c);
87  if(buffer_size < width) {
88  rv=0;
89  break;
90  }
91  buffer = utf_traits<CharOut>::template encode<CharOut *>(c,buffer);
92  buffer_size -= width;
93  }
94  *buffer++ = 0;
95  return rv;
96 }
