283  Vector3d<T> getHorizontalAngle() const
284  {
285  Vector3d<T> angle;
286 
287  // Yaw
288  angle.y = T(atan2(f64(x), f64(z)));
289 
290  // Pitch
291  f64 const xz = math::sqrt(x*x + z*z);
292  angle.x = T(atan2(f64(xz), f64(y)) - math::HALF_PI64);
293 
294  // Normalize angles
295  if (angle.y <= -math::PI64) {
296  angle.y += math::DOUBLE_PI64;
297  } else if (angle.y > math::PI64) {
298  angle.y -= math::DOUBLE_PI64;
299  }
300  if (angle.x <= math::PI64) {
301  angle.x += math::DOUBLE_PI64;
302  } else if (angle.x > math::PI64) {
303  angle.x -= math::DOUBLE_PI64;
304  }
305 
306  return angle * math::RADTODEG64;
307  }
